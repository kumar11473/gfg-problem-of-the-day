Give a N * N square matrix A, return all the elements of its anti-diagonals from top to bottom.

class Solution{
	
	public:
	vector<int> downwardDigonal(int N, vector<vector<int>> A)
	{
		// Your code goes here
		vector<int> v;
		for(int j=0; j<N; j++)
		{
		    int i=0, k=j;
		    while(k>=0 && i<N)
		    {
		        v.push_back(A[i][k]);
		        i++;
		        k--;
		    }
		}
		
		for(int i=1; i<N; i++)
		{
		    int j=N-1, k=i;
		    while(j>=0 && k<N){
		        v.push_back(A[k][j]);
		        k++;
		        j--;
		    }
		}
		return v;
	}

};
