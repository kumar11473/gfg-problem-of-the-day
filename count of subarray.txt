//User function template for C++
class Solution{
public:
	// #define ll long long

	ll countSubarray(int arr[], int n, int k) {
	    // code here
	    ll ans=0;
	    ll x=-1;
	    for(ll i=0;i<n;i++)
	    {
	        if(arr[i]>k)
	        {
	           x=i;
	           
	        }
	         ans+=x+1;
	    }
	    return ans;
	}
};
